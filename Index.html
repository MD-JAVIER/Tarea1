Ejercicios de Eventos JavaScript

1. Alternar Luz


function alternarLuz(){

    let caja = document.getElementById("caja");

    if(caja.style.backgroundColor == "yellow"){
        caja.style.backgroundColor = "gray";
    }else{
        caja.style.backgroundColor = "yellow";
    }

}

2. Contador de Clicks


let contador = 0;

document.getElementById("incrementar")
.addEventListener("click", () => {

    contador++;

    document.getElementById("contador")
    .innerText = contador;

});

3. Lista Dinámica


document.getElementById("agregar")
.addEventListener("click", () => {

    let texto = document.getElementById("item").value;

    let li = document.createElement("li");

    li.textContent = texto;

    document.getElementById("lista")
    .appendChild(li);

});

4. Cambio de Tema


document.getElementById("modo")
.addEventListener("click", () => {

    document.body.classList.toggle("dark");

});

5. Contador Regresivo


let tiempo = 10;

let intervalo = setInterval(() => {

    tiempo = tiempo - 1;

    document.getElementById("tiempo")
    .innerText = tiempo;

    if(tiempo === 0){

        clearInterval(intervalo);
    }
},1000);
